Gatling Plugin

Gatling Gradle Plugin provides simplified task & configuration to execute gatling simulations. GitHub

Using the plugins DSL:
plugins {
  id "io.devxchange.gatling" version "1.0.1"
}
Using legacy plugin application:
buildscript {
  repositories {
    maven {
      url "https://plugins.gradle.org/m2/"
    }
  }
  dependencies {
    classpath "gradle.plugin.io.devxchange.gatling:gatling-gradle-plugin:1.0.1"
  }
}

apply plugin: "io.devxchange.gatling"
